DROP TABLE IF EXISTS Member;
DROP TABLE IF EXISTS Student;
DROP TABLE IF EXISTS Staff;
			DROP TABLE IF EXISTS Topic;
DROP TABLE IF EXISTS History;

CREATE TABLE Member
(
Type varchar(10),
ID int,
PRIMARY KEY(ID)
);

CREATE TABLE Student
(
ID int,
FirstName varchar(20),
LastName varchar(20),
School varchar(100),
GradYear int,
Email varchar(30),
Gender varchar(10),
BirthDate,
FOREIGN KEY (ID) references Member(ID)
);

CREATE TABLE Staff
(
ID int,
FirstName varchar(20),
LastName varchar(20),
School varchar(100),
Email varchar(30),
Gender varchar(10),
BirthDate,
FOREIGN KEY (ID) references Member(ID)
);

			CREATE TABLE Topic
			(
			MarketPlace varchar(50),
			Announcements varchar(50),
			ClassRegistration varchar(50)
			);

CREATE TABLE History
(
ID int,
FirstName varchar(20),
LastName varchar(20),
Type varchar(10),
Topic varchar(50),
CommentDate Date,
Comment varchar(8000),
RecordNumber int auto_increment,
FOREIGN KEY (ID) references Member(ID)
);
