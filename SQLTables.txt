DROP TABLE IF EXISTS Student;
DROP TABLE IF EXISTS Staff;
DROP TABLE IF EXISTS History;

CREATE TABLE Member
(
Position varchar(10),
ID int,
PRIMARY KEY(ID)
);


CREATE TABLE Student
(
ID int,
FirstName varchar(20),
LastName varchar(20),
School varchar(100),
GradYear int,
Email varchar(30),
Gender varchar(10),
BirthDate Date,
Password varchar(30),
FOREIGN KEY (ID) references Member(ID)
);


CREATE TABLE Staff
(
ID int,
FirstName varchar(20),
LastName varchar(20),
School varchar(100),
Email varchar(30),
Gender varchar(10),
BirthDate Date,
Password varchar(30),
FOREIGN KEY (ID) references Member(ID)
);


CREATE TABLE History
(
ID int,
Topic varchar(50),
CommentDate Date,
Comment varchar(8000),
RecordNumber int,
FOREIGN KEY (ID) references Member(ID)
);
